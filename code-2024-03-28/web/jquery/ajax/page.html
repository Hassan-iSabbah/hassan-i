$(document).ready(function () {
// Initially, load the short content
loadContent('short-content.txt');

// Handle "Read more" button click
$('#read-more-btn').on('click', function () {
// Load the long content when the button is clicked
loadContent('long-content.txt');
});
});

function loadContent(filename) {
$.ajax({
url: './ajax/' + filename,
type: 'GET',
dataType: 'text',
success: function (data) {
$('#content').html('<p id="short-content">' + data + '</p>');
},
error: function () {
console.error('Error loading content.');
}
});
}
